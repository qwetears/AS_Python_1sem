def rle_compress(s):
    compressed = []
    length = len(s)
    max_savings = 0
    max_char = None

    i = 0
    while i < length:
        count = 1
        while i + 1 < length and s[i] == s[i + 1]:
            count += 1
            i += 1
        
        compressed.append(s[i] + str(count))
        
        original_length = count
        compressed_length = len(s[i] + str(count))
        savings = original_length - compressed_length

        if savings > max_savings or (savings == max_savings and (max_char is None or s[i] < max_char)):
            max_savings = savings
            max_char = s[i]

        i += 1

    compressed_string = ''.join(compressed)
    compressed_length = len(compressed_string)

    return length, compressed_length, max_char

s = input("Введите строку: ")

original_length, compressed_length, max_saving_char = rle_compress(s)

print(f"Длина исходной строки: {original_length}")
print(f"Длина сжатой строки: {compressed_length}")
if max_saving_char is not None:
    print(f"Символ, давший максимальную 'экономию': {max_saving_char}")
else:
    print("Нет символов, давших экономию.")
